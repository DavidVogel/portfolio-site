106 void SallenKeyFilter::process12(const poly_float* audio_in, int num_samples,
107 poly_float current_resonance,
108 poly_float current_drive, poly_float current_post_multiply,
109 poly_float current_low, poly_float current_band, poly_float current_high) {
110 mono_float tick_increment = 1.0f / num_samples;
111 // Compute deltas for parameter smoothing
112 poly_float delta_resonance = (resonance_ - current_resonance) * tick_increment;
113 poly_float delta_drive = (drive_ - current_drive) * tick_increment;
114 poly_float delta_post_multiply = (post_multiply_ - current_post_multiply) * tick_increment;
115 poly_float delta_low = (low_pass_amount_ - current_low) * tick_increment;
116 poly_float delta_band = (band_pass_amount_ - current_band) * tick_increment;
117 poly_float delta_high = (high_pass_amount_ - current_high) * tick_increment;
118
119 // Retrieve coefficient lookup and MIDI-based cutoff data
120 const CoefficientLookup* coefficient_lookup = getCoefficientLookup();
121 const poly_float* midi_cutoff_buffer = filter_state_.midi_cutoff_buffer;
122 poly_float base_midi = midi_cutoff_buffer[num_samples - 1];
123 poly_float base_frequency =
124 utils::midiNoteToFrequency(base_midi) * (1.0f / getSampleRate());
125
126 poly_float* audio_out = output()->buffer;
127
128 // Main loop
129 for (int i = 0; i < num_samples; ++i) {
130 // Calculate the local cutoff coefficient
131 poly_float midi_delta = midi_cutoff_buffer[i] - base_midi;
132 poly_float frequency =
133 utils::min(base_frequency * futils::midiOffsetToRatio(midi_delta), 1.0f);
134 poly_float coefficient = coefficient_lookup->cubicLookup(frequency);
135
136 // Update parameters
137 current_resonance += delta_resonance;
138 current_drive += delta_drive;
139 current_post_multiply += delta_post_multiply;
140 current_low += delta_low;
141 current_band += delta_band;
142 current_high += delta_high;
143
144 // Compute resonance and other Sallen-Key factors
145 poly_float coefficient2 = coefficient * 2.0f;
146 poly_float resonance = tuneResonance(current_resonance, coefficient2);
147 poly_float stage1_feedback_mult = coefficient2 - coefficient * coefficient - 1.0f;
148 poly_float normalizer =
149 poly_float(1.0f) / (resonance * (coefficient * coefficient - coefficient) + 1.0f);
150
151 // Tick the filter (2-pole version)
152 tick(audio_in[i], coefficient, resonance, stage1_feedback_mult, current_drive, normalizer);
153
154 // 2-pole Sallen-Key output retrieval
155 poly_float stage2_input = stage1_.getCurrentState();
156 poly_float low_pass = stage2_.getCurrentState();
157 poly_float band_pass = stage2_input - low_pass;
158 poly_float high_pass = stage1_input_ - stage2_input - band_pass;
159
160 // Mix the final output based on low, band, and high amounts
161 poly_float low = current_low * low_pass;
162 poly_float band_low = utils::mulAdd(low, current_band, band_pass);
163 audio_out[i] =
164 utils::mulAdd(band_low, current_high, high_pass) * current_post_multiply;
165
166 VITAL_ASSERT(utils::isFinite(audio_out[i]));
167 }

320 void SallenKeyFilter::setupFilter(const FilterState& filter_state) {
321 // Convert MIDI cutoff to frequency
322 cutoff_ = utils::midiNoteToFrequency(filter_state.midi_cutoff);
323 float min_nyquist = getSampleRate() * kMinNyquistMult;
324 cutoff_ = utils::clamp(cutoff_, kMinCutoff, min_nyquist);
325
326 // Compute resonance from the input parameter
327 poly_float resonance_percent =
328 utils::clamp(filter_state.resonance_percent, 0.0f, 1.0f);
329 // Use sqrt to skew resonance distribution
330 resonance_percent = utils::sqrt(resonance_percent);
331
332 resonance_ =
333 utils::interpolate(kMinResonance, kMaxResonance, resonance_percent);
334 // Additional boost based on drive
335 resonance_ += filter_state.drive_percent
336 * filter_state.resonance_percent
337 * kDriveResonanceBoost;
338
339 // pass_blend mapped to -1..1
340 poly_float blend =
341 utils::clamp(filter_state.pass_blend - 1.0f, -1.0f, 1.0f);
342
343 // Scale drive by the squared resonance factor
344 poly_float resonance_scale =
345 resonance_percent * resonance_percent * 2.0f + 1.0f;
346 drive_ = poly_float(filter_state.drive) / resonance_scale;
347
348 // Handle style-specific parameter routing
349 if (filter_state.style == kDualNotchBand) {
350 // Specialized blending for dual-notch
351 poly_float t = blend * 0.5f + 0.5f;
352 poly_float drive_t = poly_float::min(-blend + 1.0f, 1.0f);
353 poly_float drive_mult = -t + 2.0f;
354 drive_ = utils::interpolate(filter_state.drive, drive_ * drive_mult, drive_t);
355
356 low_pass_amount_ = t;
357 band_pass_amount_ = 0.0f;
358 high_pass_amount_ = 1.0f;
359 }
360 else if (filter_state.style == kNotchPassSwap) {
361 // Blend for a notch/pass swap style
362 poly_float drive_t = poly_float::abs(blend);
363 drive_ = utils::interpolate(filter_state.drive, drive_, drive_t);
364
365 low_pass_amount_ = utils::min(-blend + 1.0f, 1.0f);
366 band_pass_amount_ = 0.0f;
367 high_pass_amount_ = utils::min(blend + 1.0f, 1.0f);
368 }
369 else if (filter_state.style == kBandPeakNotch) {
370 // Band/peak/notch style
371 poly_float drive_t = poly_float::min(-blend + 1.0f, 1.0f);
372 drive_ = utils::interpolate(filter_state.drive, drive_, drive_t);
373
374 poly_float drive_inv_t = -drive_t + 1.0f;
375 poly_float mult =
376 utils::sqrt((drive_inv_t * drive_inv_t) * 0.5f + 0.5f);
377 poly_float peak_band_value = -utils::max(-blend, 0.0f);
378 low_pass_amount_ = mult * (peak_band_value + 1.0f);
379 band_pass_amount_ =
380 mult * (peak_band_value - blend + 1.0f) * 2.0f;
381 high_pass_amount_ = low_pass_amount_;
382 }
383 else {
384 // Default 12 dB or 24 dB blending
385 band_pass_amount_ = utils::sqrt(-blend * blend + 1.0f);
386 poly_mask blend_mask = poly_float::lessThan(blend, 0.0f);
387 low_pass_amount_ = (-blend) & blend_mask;
388 high_pass_amount_ = blend & ~blend_mask;
389 }
390
391 // Post-multiply factor
392 post_multiply_ =
393 poly_float(1.0f) / utils::sqrt(resonance_scale * drive_);
394 }

409 force_inline void SallenKeyFilter::tick24(poly_float audio_in, poly_float coefficient,
410 poly_float resonance, poly_float stage1_feedback_mult, poly_float drive,
411 poly_float pre_normalizer, poly_float normalizer,
412 poly_float low, poly_float band, poly_float high) {
413 // Preliminary stage feedback using the pre_stage1_ and pre_stage2_ filters
414 poly_float mult_stage2 = -coefficient + 1.0f;
415 poly_float feedback =
416 utils::mulAdd(stage1_feedback_mult * pre_stage1_.getNextState(),
417 mult_stage2, pre_stage2_.getNextState());
418
419 // Pre-stage input
420 poly_float stage1_input = (audio_in - feedback) * pre_normalizer;
421
422 // Pass through two one-pole filters for preliminary 2 poles
423 poly_float stage1_out = pre_stage1_.tickBasic(stage1_input, coefficient);
424 poly_float stage2_out = pre_stage2_.tickBasic(stage1_out, coefficient);
425
426 // Temporary band-pass and high-pass from these first 2 poles
427 poly_float band_pass_out = stage1_out - stage2_out;
428 poly_float high_pass_out = stage1_input - stage1_out - band_pass_out;
429
430 // Combine them based on the users low, band, high
431 poly_float low_out = low * stage2_out;
432 poly_float band_low_out = utils::mulAdd(low_out, band, band_pass_out);
433 poly_float audio_out = utils::mulAdd(band_low_out, high, high_pass_out);
434
435 // The second pair of poles uses the main tick() function
436 tick(audio_out, coefficient, resonance, stage1_feedback_mult, drive, normalizer);
437 }

448 force_inline void SallenKeyFilter::tick(poly_float audio_in, poly_float coefficient, poly_float resonance,
449 poly_float stage1_feedback_mult, poly_float drive, poly_float normalizer) {
450 // Compute feedback from the first stage
451 poly_float mult_stage2 = -coefficient + 1.0f;
452 poly_float feedback =
453 utils::mulAdd(stage1_feedback_mult * stage1_.getNextState(),
454 mult_stage2, stage2_.getNextState());
455
456 // Nonlinear saturation (tanh) is used here to manage extremes
457 stage1_input_ = futils::tanh((drive * audio_in - resonance * feedback) * normalizer);
458
459 // Pass through two one-pole filters (1st and 2nd stage) for 2-pole effect
460 poly_float stage1_out = stage1_.tickBasic(stage1_input_, coefficient);
461 stage2_.tickBasic(stage1_out, coefficient);
