130 void LinkwitzRileyFilter::computeCoefficients() {
131 // Warp frequency using bilinear transform
132 mono_float warp = 1.0f / tanf(kPi * cutoff_ / getSampleRate());
133 mono_float warp2 = warp * warp;
134 mono_float mult = 1.0f / (1.0f + kSqrt2 * warp + warp2);
135
136 // Low-pass filter coefficients
137 low_in_0_ = mult;
138 low_in_1_ = 2.0f * mult;
139 low_in_2_ = mult;
140 low_out_1_ = -2.0f * (1.0f - warp2) * mult;
141 low_out_2_ = -(1.0f - kSqrt2 * warp + warp2) * mult;
142
143 // High-pass filter coefficients
144 high_in_0_ = warp2 * mult;
145 high_in_1_ = -2.0f * high_in_0_;
146 high_in_2_ = high_in_0_;
147 high_out_1_ = low_out_1_;
148 high_out_2_ = low_out_2_;
149 }
