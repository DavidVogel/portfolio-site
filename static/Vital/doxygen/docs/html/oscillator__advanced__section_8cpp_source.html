272 void render(OpenGlWrapper& open_gl, bool animate) override {
273 if (voices_slider_ == nullptr)
274 return;
275
276 int voices = getOutputsTotal(voices_outputs_, voices_slider_->getValue(), animate)[0];
277 voices = std::max(voices, 0);
278 if (voices <= 2)
279 return;
280
281 vital::poly_float frame = getOutputsTotal(wave_frame_outputs_, wave_frame_slider_->getValue(), animate);
282 vital::poly_float morph = getOutputsTotal(spectral_morph_outputs_, spectral_morph_slider_->getValue(), animate);
283 vital::poly_float distortion = getOutputsTotal(distortion_outputs_, distortion_slider_->getValue(), animate);
284
285 vital::poly_float frame_spread = getOutputsTotal(table_spread_outputs_,
286 table_spread_slider_->getValue(), animate);
287 vital::poly_float morph_spread = getOutputsTotal(spectral_spread_outputs_,
288 spectral_spread_slider_->getValue(), animate);
289 vital::poly_float distortion_spread = getOutputsTotal(distortion_spread_outputs_,
290 distortion_spread_slider_->getValue(), animate);
291
292 setColor(findColour(Skin::kWidgetSecondary1, true));
293 setBarWidth(2.0f / getWidth());
294
295 float height_buffer = 2.0f * findValue(Skin::kWidgetMargin) / getHeight();
296 float height = (2.0f - height_buffer) / kNumSpreads - height_buffer;
297 float y = height_buffer - 1.0f;
298 for (int s = 0; s < kNumSpreads; ++s) {
299 int start = s * vital::SynthOscillator::kMaxUnison;
300 for (int i = 0; i < vital::SynthOscillator::kMaxUnison; ++i) {
301 setBottom(start + i, y + height);
302 setY(start + i, y);
303 }
304
305 y += height + height_buffer;
306 }
307
308 float buffer = 2.0f * findValue(Skin::kWidgetMargin) / getWidth();
309 float mult = 2.0f - 2.0f * buffer;
310 float offset = -1.0f + buffer - 1.0f / getWidth();
311 float frame_scale = 1.0f / (vital::kNumOscillatorWaveFrames - 1.0f);
312 voices += voices % 2;
313 for (int i = 0; i < voices; i += 2) {
314 float t = 2.0f * i / (voices - 2.0f);
315 vital::poly_float voice_frame = frame + frame_spread * t;
316 voice_frame = vital::utils::clamp(voice_frame * frame_scale, 0.0f, 1.0f);
317 vital::poly_float voice_morph = vital::utils::clamp(morph + morph_spread * t, 0.0f, 1.0f);
318 vital::poly_float voice_distortion = vital::utils::clamp(distortion + distortion_spread * t, 0.0f, 1.0f);
319
320 setX(i, voice_frame[0] * mult + offset);
321 setX(i + 1, voice_frame[1] * mult + offset);
322
323 setX(i + vital::SynthOscillator::kMaxUnison, voice_morph[0] * mult + offset);
324 setX(i + vital::SynthOscillator::kMaxUnison + 1, voice_morph[1] * mult + offset);
325
326 setX(i + 2 * vital::SynthOscillator::kMaxUnison, voice_distortion[0] * mult + offset);
327 setX(i + 2 * vital::SynthOscillator::kMaxUnison + 1, voice_distortion[1] * mult + offset);
328 }
329
330 for (int s = 0; s < kNumSpreads; ++s) {
331 int start = s * vital::SynthOscillator::kMaxUnison;
332 for (int i = voices; i < vital::SynthOscillator::kMaxUnison; ++i)
333 setX(start + i, -2.0f);
334 }
335
336 BarRenderer::render(open_gl, animate);
337 }
