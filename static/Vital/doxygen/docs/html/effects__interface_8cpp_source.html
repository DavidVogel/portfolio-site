167void EffectsInterface::setEffectPositions() {
168 if (getWidth() <= 0 || getHeight() <= 0)
169 return;
170
171 int padding = getPadding();
172 int large_padding = findValue(Skin::kLargePadding);
173 int shadow_width = getComponentShadowWidth();
174 int start_x = effect_order_->getRight() + large_padding;
175 int effect_width = getWidth() - start_x - large_padding;
176 int knob_section_height = getKnobSectionHeight();
177 int widget_margin = findValue(Skin::kWidgetMargin);
178 int effect_height = 2 * knob_section_height - widget_margin;
179 int y = 0;
180
181 Point<int> position = viewport_.getViewPosition();
182
183 for (int i = 0; i < vital::constants::kNumEffects; ++i) {
184 bool enabled = effect_order_->effectEnabled(i);
185 effects_list_[effect_order_->getEffectIndex(i)]->setVisible(enabled);
186 if (enabled) {
187 effects_list_[effect_order_->getEffectIndex(i)]->setBounds(shadow_width, y, effect_width, effect_height);
188 y += effect_height + padding;
189 }
190 }
191
192 container_->setBounds(0, 0, viewport_.getWidth(), y - padding);
193 viewport_.setViewPosition(position);
194
195 for (Listener* listener : listeners_)
196 listener->effectsMoved();
197
198 container_->setScrollWheelEnabled(container_->getHeight() <= viewport_.getHeight());
199 setScrollBarRange();
200 repaintBackground();
201}

208void EffectsInterface::renderOpenGlComponents(OpenGlWrapper& open_gl, bool animate) {
209 ScopedLock lock(open_gl_critical_section_);
210
211 OpenGlComponent::setViewPort(&viewport_, open_gl);
212
213 float image_width = vital::utils::nextPowerOfTwo(background_.getImageWidth());
214 float image_height = vital::utils::nextPowerOfTwo(background_.getImageHeight());
215 int mult = getPixelMultiple();
216 float width_ratio = image_width / (container_->getWidth() * mult);
217 float height_ratio = image_height / (viewport_.getHeight() * mult);
218 float y_offset = (2.0f * viewport_.getViewPositionY()) / getHeight();
219
220 background_.setTopLeft(-1.0f, 1.0f + y_offset);
221 background_.setTopRight(-1.0 + 2.0f * width_ratio, 1.0f + y_offset);
222 background_.setBottomLeft(-1.0f, 1.0f - 2.0f * height_ratio + y_offset);
223 background_.setBottomRight(-1.0 + 2.0f * width_ratio, 1.0f - 2.0f * height_ratio + y_offset);
224
225 background_.setColor(Colours::white);
226 background_.drawImage(open_gl);
227 SynthSection::renderOpenGlComponents(open_gl, animate);
228}
