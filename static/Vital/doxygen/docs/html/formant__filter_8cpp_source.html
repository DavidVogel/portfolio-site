161 void FormantFilter::init() {
162 static const cr::Value k12Db(DigitalSvf::k12Db);
163
164 // Center value for Spread interpolation around kCenterMidi
165 Value* center = new Value(kCenterMidi);
166 addIdleProcessor(center);
167
168 for (int i = 0; i < kNumFormants; ++i) {
169 // Create interpolation nodes for gain, Q, and MIDI cutoff
170 cr::BilinearInterpolate* formant_gain = new cr::BilinearInterpolate();
171 cr::BilinearInterpolate* formant_q = new cr::BilinearInterpolate();
172 BilinearInterpolate* formant_midi = new BilinearInterpolate();
173
174 // Plug in the corner (top-left, top-right, bottom-left, bottom-right) values
175 for (int p = 0; p < kNumFormantPositions; ++p) {
176 formant_gain->plug(&formant_styles[style_][p][i].gain,
177 cr::BilinearInterpolate::kPositionStart + p);
178 formant_q->plug(&formant_styles[style_][p][i].resonance,
179 cr::BilinearInterpolate::kPositionStart + p);
180 formant_midi->plug(&formant_styles[style_][p][i].midi_cutoff,
181 BilinearInterpolate::kPositionStart + p);
182 }
183
184 // Connect X and Y interpolation inputs
185 formant_gain->useInput(input(kInterpolateX), cr::BilinearInterpolate::kXPosition);
186 formant_q->useInput(input(kInterpolateX), cr::BilinearInterpolate::kXPosition);
187 formant_midi->useInput(input(kInterpolateX), BilinearInterpolate::kXPosition);
188
189 formant_gain->useInput(input(kInterpolateY), cr::BilinearInterpolate::kYPosition);
190 formant_q->useInput(input(kInterpolateY), cr::BilinearInterpolate::kYPosition);
191 formant_midi->useInput(input(kInterpolateY), BilinearInterpolate::kYPosition);
192
193 // Interpolate node for spreading the formant MIDI cutoff around a center
194 Interpolate* formant_midi_spread = new Interpolate();
195 formant_midi_spread->useInput(input(kSpread), Interpolate::kFractional);
196 formant_midi_spread->useInput(input(kReset), Interpolate::kReset);
197 formant_midi_spread->plug(center, Interpolate::kTo);
198 formant_midi_spread->plug(formant_midi, Interpolate::kFrom);
199
200 // Offset (transpose) the MIDI cutoff
201 Add* formant_midi_adjust = new Add();
202 formant_midi_adjust->useInput(input(kTranspose), 0);
203 formant_midi_adjust->plug(formant_midi_spread, 1);
204
205 // Multiply the resonance by an additional user-supplied resonance factor
206 cr::Multiply* formant_q_adjust = new cr::Multiply();
207 formant_q_adjust->useInput(input(kResonance), 0);
208 formant_q_adjust->plug(formant_q, 1);
209
210 // Connect the formant manager's DigitalSvf for this formant
211 formant_manager_->getFormant(i)->useInput(input(kAudio), DigitalSvf::kAudio);
212 formant_manager_->getFormant(i)->useInput(input(kReset), DigitalSvf::kReset);
213 formant_manager_->getFormant(i)->plug(&k12Db, DigitalSvf::kStyle);
214 formant_manager_->getFormant(i)->plug(&constants::kValueOne, DigitalSvf::kPassBlend);
215 formant_manager_->getFormant(i)->plug(formant_gain, DigitalSvf::kGain);
216 formant_manager_->getFormant(i)->plug(formant_q_adjust, DigitalSvf::kResonance);
217 formant_manager_->getFormant(i)->plug(formant_midi_adjust, DigitalSvf::kMidiCutoff);
218
219 // Add these processors to the chain
220 addProcessor(formant_gain);
221 addProcessor(formant_q);
222 addProcessor(formant_q_adjust);
223 addProcessor(formant_midi);
224 addProcessor(formant_midi_spread);
225 addProcessor(formant_midi_adjust);
226 }
227
228 // Finish initialization of the ProcessorRouter
229 ProcessorRouter::init();

236 void FormantFilter::setupFilter(const FilterState& filter_state) {
237 // Clamp the style to a valid index
238 int style = std::min(filter_state.style, FormantFilter::kNumFormantStyles - 1);
239
240 for (int i = 0; i < vital::kNumFormants; ++i) {
241 // Interpolate the formant parameters from the style's corner values
242 FilterState formant_setting =
243 interpolateFormants(formant_styles[style][kTopLeft][i],
244 formant_styles[style][kTopRight][i],
245 formant_styles[style][kBottomLeft][i],
246 formant_styles[style][kBottomRight][i],
247 filter_state.interpolate_x,
248 filter_state.interpolate_y);
249
250 // Blend between the formant's MIDI cutoff and the global center
251 formant_setting.midi_cutoff = utils::interpolate(formant_setting.midi_cutoff,
252 kCenterMidi,
253 filter_state.pass_blend);
254
255 // Add global transpose
256 formant_setting.midi_cutoff += filter_state.transpose;
257
258 // Scale resonance by global resonance_percent
259 formant_setting.resonance_percent *= filter_state.resonance_percent;
260
261 // Use 12dB style for each formant
262 formant_setting.style = DigitalSvf::k12Db;
263 formant_setting.pass_blend = 1.0f;
264
265 // Apply to the underlying DigitalSvf
266 vital::DigitalSvf* formant = formant_manager_->getFormant(i);
267 formant->setupFilter(formant_setting);
268 }
