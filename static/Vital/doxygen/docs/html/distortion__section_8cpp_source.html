35 void drawDistortion(OpenGlWrapper& open_gl, bool animate, int index) {
36 vital::poly_float drive = vital::Distortion::getDriveValue(type_slider_->getValue(), getDrive());
37
38 float width = getWidth();
39 float height = getHeight();
40 float y_scale = height / 2.0f;
41 int num_points = numPoints();
42 for (int i = 0; i < num_points; ++i) {
43 float t = i / (num_points - 1.0f);
44 float val = 2.0f * t - 1.0f;
45 setXAt(i, t * width);
46 float result = kDrawPercent * vital::Distortion::getDrivenValue(type_slider_->getValue(), val, drive)[index];
47 setYAt(i, (1.0f - result) * y_scale);
48 }
49
50 OpenGlLineRenderer::render(open_gl, animate);
51 }

135void DistortionFilterResponse::init(OpenGlWrapper& open_gl) {
136 OpenGlLineRenderer::init(open_gl);
137
138 const GLchar* varyings[] = { "response_out" };
139 open_gl.context.extensions.glGenVertexArrays(1, &vertex_array_object_);
140 open_gl.context.extensions.glBindVertexArray(vertex_array_object_);
141
142 GLsizeiptr data_size = static_cast<GLsizeiptr>(kResolution * sizeof(float));
143 open_gl.context.extensions.glGenBuffers(1, &line_buffer_);
144 open_gl.context.extensions.glBindBuffer(GL_ARRAY_BUFFER, line_buffer_);

149 open_gl.context.extensions.glBufferData(GL_ARRAY_BUFFER, data_size, nullptr, GL_STATIC_READ);

152 Shaders::kColorFragment, varyings);
153 response_shader_.shader = shader;
154
155 shader->use();
156 response_shader_.position = getAttribute(open_gl, *shader, "position");
157
158 response_shader_.mix = getUniform(open_gl, *shader, "mix");
159 response_shader_.midi_cutoff = getUniform(open_gl, *shader, "midi_cutoff");
160 response_shader_.resonance = getUniform(open_gl, *shader, "resonance");
161 response_shader_.drive = getUniform(open_gl, *shader, "drive");
162 response_shader_.db24 = getUniform(open_gl, *shader, "db24");
163
164 for (int s = 0; s < FilterResponseShader::kMaxStages; ++s) {
165 String stage = String("stage") + String(s);
166 response_shader_.stages[s] = getUniform(open_gl, *shader, stage.toRawUTF8());

210void DistortionFilterResponse::loadShader(int index) {
211 filter_.setupFilter(filter_state_);
212 response_shader_.shader->use();
213 float min_cutoff = cutoff_slider_->getMinimum() + 0.001;
214 float cutoff = std::max(min_cutoff, filter_state_.midi_cutoff[index]);
215 response_shader_.midi_cutoff->set(cutoff);
216
217 float resonance = vital::utils::clamp(filter_.getResonance()[index], 0.0f, 2.0f);
218 response_shader_.resonance->set(resonance);
219 response_shader_.mix->set(1.0f);
220
221 response_shader_.drive->set(filter_.getDrive()[index]);
222 response_shader_.db24->set(filter_state_.style != vital::SynthFilter::k12Db ? 1.0f : 0.0f);
223
224 response_shader_.stages[0]->set(filter_.getLowAmount()[index]);
225 response_shader_.stages[1]->set(filter_.getBandAmount()[index]);
226 response_shader_.stages[2]->set(filter_.getHighAmount()[index]);
227 response_shader_.stages[3]->set(filter_.getLowAmount24(filter_state_.style)[index]);
228 response_shader_.stages[4]->set(filter_.getHighAmount24(filter_state_.style)[index]);
229}

231void DistortionFilterResponse::bind(OpenGLContext& open_gl_context) {
232 open_gl_context.extensions.glBindVertexArray(vertex_array_object_);

250void DistortionFilterResponse::renderLineResponse(OpenGlWrapper& open_gl) {
251 open_gl.context.extensions.glBeginTransformFeedback(GL_POINTS);
252 glDrawArrays(GL_POINTS, 0, kResolution);
253 open_gl.context.extensions.glEndTransformFeedback();
254
255 void* buffer = open_gl.context.extensions.glMapBufferRange(GL_TRANSFORM_FEEDBACK_BUFFER, 0,
256 kResolution * sizeof(float), GL_MAP_READ_BIT);
257
258 float* response_data = (float*)buffer;
259 float x_adjust = getWidth();
260 float y_adjust = getHeight() / 2.0f;
261 for (int i = 0; i < kResolution; ++i) {
262 setXAt(i, x_adjust * i / (kResolution - 1.0f));
263 setYAt(i, y_adjust * (1.0 - response_data[i]));
264 }
265
266 open_gl.context.extensions.glUnmapBuffer(GL_TRANSFORM_FEEDBACK_BUFFER);
267}

269void DistortionFilterResponse::drawFilterResponse(OpenGlWrapper& open_gl, bool animate) {
270 setupFilterState();
271 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
272 glEnable(GL_BLEND);
273 setViewPort(open_gl);
274
275 Colour color_line = findColour(Skin::kWidgetPrimary2, true);
276 Colour color_fill_to = findColour(Skin::kWidgetSecondary2, true);
277 float fill_fade = findValue(Skin::kWidgetFillFade);
278 Colour color_fill_from = color_fill_to.withMultipliedAlpha(1.0f - fill_fade);
279
280 setLineWidth(findValue(Skin::kWidgetLineWidth));
281 setFillCenter(findValue(Skin::kWidgetFillCenter));
282 if (active_) {
283 bind(open_gl.context);
284 loadShader(1);
285 renderLineResponse(open_gl);
286
287 setFillColors(color_fill_from, color_fill_to);
288 setColor(color_line);
289 OpenGlLineRenderer::render(open_gl, animate);
290 }
291
292 glEnable(GL_BLEND);
293 color_line = findColour(Skin::kWidgetPrimary1, true);
294 color_fill_to = findColour(Skin::kWidgetSecondary1, true);
295 if (!active_) {
296 color_line = findColour(Skin::kWidgetPrimaryDisabled, true);
297 color_fill_to = findColour(Skin::kWidgetSecondaryDisabled, true);
298 }
299 color_fill_from = color_fill_to.withMultipliedAlpha(1.0f - fill_fade);
300
301 bind(open_gl.context);
302 loadShader(0);
303 renderLineResponse(open_gl);
304
305 setFillColors(color_fill_from, color_fill_to);
306 setColor(color_line);
307 OpenGlLineRenderer::render(open_gl, animate);
308
