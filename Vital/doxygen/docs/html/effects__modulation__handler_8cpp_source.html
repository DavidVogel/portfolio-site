233 void EffectsModulationHandler::createArticulation() {
234 // Velocity tracking
235 Output* velocity_track_amount = createPolyModControl("velocity_track");
236 cr::Interpolate* velocity_track_mult = new cr::Interpolate();
237 velocity_track_mult->plug(&constants::kValueOne, Interpolate::kFrom);
238 velocity_track_mult->plug(velocity(), Interpolate::kTo);
239 velocity_track_mult->plug(velocity_track_amount, Interpolate::kFractional);
240 addProcessor(velocity_track_mult);
241
242 // Portamento controls
243 Output* portamento = createPolyModControl("portamento_time");
244 Output* portamento_slope = createPolyModControl("portamento_slope");
245 Value* portamento_force = createBaseControl("portamento_force");
246 Value* portamento_scale = createBaseControl("portamento_scale");
247
248 current_midi_note_ = new PortamentoSlope();
249 current_midi_note_->plug(last_note(), PortamentoSlope::kSource);
250 current_midi_note_->plug(note(), PortamentoSlope::kTarget);
251 current_midi_note_->plug(portamento_force, PortamentoSlope::kPortamentoForce);
252 current_midi_note_->plug(portamento_scale, PortamentoSlope::kPortamentoScale);
253 current_midi_note_->plug(portamento, PortamentoSlope::kRunSeconds);
254 current_midi_note_->plug(portamento_slope, PortamentoSlope::kSlopePower);
255 current_midi_note_->plug(voice_event(), PortamentoSlope::kReset);
256 current_midi_note_->plug(note_pressed(), PortamentoSlope::kNumNotesPressed);
257 setVoiceMidi(current_midi_note_->output());
258 addProcessor(current_midi_note_);
259
260 // Pitch bend
261 Output* pitch_bend_range = createPolyModControl("pitch_bend_range");
262 Output* voice_tune = createPolyModControl("voice_tune");
263 cr::Multiply* pitch_bend = new cr::Multiply();
264 pitch_bend->plug(pitch_wheel(), 0);
265 pitch_bend->plug(pitch_bend_range, 1);
266
267 bent_midi_ = new cr::VariableAdd();
268 bent_midi_->plugNext(current_midi_note_);
269 bent_midi_->plugNext(pitch_bend);
270 bent_midi_->plugNext(local_pitch_bend());
271 bent_midi_->plugNext(voice_tune);
272
273 static const cr::Value max_midi_invert(1.0f / (kMidiSize - 1));
274 note_percentage_ = new cr::Multiply();
275 note_percentage_->plug(&max_midi_invert, 0);
276 note_percentage_->plug(bent_midi_, 1);
277 addProcessor(note_percentage_);
278
279 static const cr::Value reference_adjust(-kMidiTrackCenter);
280 note_from_reference_->plug(&reference_adjust, 0);
281 note_from_reference_->plug(bent_midi_, 1);
282 addProcessor(note_from_reference_);
283
284 addProcessor(pitch_bend);
285 addProcessor(bent_midi_);
286 }

293 void EffectsModulationHandler::process(int num_samples) {
294 poly_mask reset_mask = reset()->trigger_mask;
295 if (reset_mask.anyMask())
296 resetFeedbacks(reset_mask);
297
298 VoiceHandler::process(num_samples);
299 note_retriggered_.clearTrigger();
300
301 // If no notes are active, clear status outputs
302 if (getNumActiveVoices() == 0) {
303 for (auto& status_source : data_->status_outputs)
304 status_source.second->clear();
305 }
306 // Otherwise, update active connections
307 else {
308 poly_mask voice_mask = getCurrentVoiceMask();
309 for (int i = 0; i < vital::kMaxModulationConnections; ++i) {
310 ModulationConnectionProcessor* processor = modulation_bank_.atIndex(i)->modulation_processor.get();
311 if (processor->enabled()) {
312 poly_float* buffer = processor->output()->buffer;
313 poly_float masked_value = buffer[0] & voice_mask;
314 buffer[0] = masked_value + utils::swapVoices(masked_value);
315 }
316 }
317 for (auto& status_source : data_->status_outputs)
318 status_source.second->update(voice_mask);
319 }
320 }
