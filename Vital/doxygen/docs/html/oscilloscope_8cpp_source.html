76void Spectrogram::applyWindow() {
77 // Applies a Hann window (approximated by a cosine transform)
78 static constexpr double kRadianIncrement = vital::kPi / (kAudioSize - 1.0);
79
80 double real = -1.0f;
81 double imag = 0.0f;
82 double real_mult = cos(kRadianIncrement);
83 double imag_mult = sin(kRadianIncrement);
84
85 for (int i = 0; i < kAudioSize; ++i) {
86 transform_buffer_[i] *= 0.5f * (real + 1.0f);
87
88 double next_real = real * real_mult - imag * imag_mult;
89 double next_imag = imag * real_mult + real * imag_mult;
90 real = next_real;
91 imag = next_imag;
92 }
93}

95void Spectrogram::updateAmplitudes(int index, int offset) {
96 static constexpr float kMinAmp = 0.000001f;
97 static constexpr float kStartScaleAmp = 0.001f;
98 static constexpr float kMinDecay = 0.06f;
99
100 if (memory_ == nullptr)
101 return;
102
103 float min_frequency = min_frequency_ / oversample_amount_;
104 float max_frequency = max_frequency_ / oversample_amount_;
105 float* amps = index == 0 ? left_amps_ : right_amps_;
106 float sample_hz = (1.0f * sample_rate_) / kAudioSize;
107 float start_octave = log2f(min_frequency / sample_hz);
108 float end_octave = std::min<float>(kBits - start_octave + 1.0f, log2f(max_frequency / sample_hz));
109 float num_octaves = end_octave - start_octave;
110
111 memory_->readSamples(transform_buffer_, kAudioSize, offset, index);
112 applyWindow();
113 transform_.transformRealForward(transform_buffer_);
114 std::complex<float>* frequency_data = (std::complex<float>*)transform_buffer_;
115
116 float last_bin = powf(2.0f, start_octave);
117 for (int i = 0; i < kResolution; ++i) {
118 float t = i / (kResolution - 1.0f);
119 float octave = num_octaves * t + start_octave;
120 float bin = powf(2.0f, octave);
121
122 int bin_index = last_bin;
123 float bin_t = last_bin - bin_index;
124 float prev_amplitude = std::abs(frequency_data[bin_index]);
125 float next_amplitude = std::abs(frequency_data[bin_index + 1]);
126 float amplitude = vital::utils::interpolate(prev_amplitude, next_amplitude, bin_t);
127 if (bin - last_bin > 1.0f) {
128 for (int j = last_bin + 1; j < bin; ++j)
129 amplitude = std::max(amplitude, std::abs(frequency_data[j]));
130 }
131 last_bin = bin;
132
133 amplitude = std::max(kMinAmp, 2.0f * amplitude / kAudioSize);
134 float db = vital::utils::magnitudeToDb(std::max(amplitude, amps[i]) / kStartScaleAmp);
135 db += octave * kDbSlopePerOctave;
136 float decay = std::max(kMinDecay, std::min(1.0f, kDecayMult * db));
137 amps[i] = std::max(kMinAmp, vital::utils::interpolate(amps[i], amplitude, decay));
138 }
139}
