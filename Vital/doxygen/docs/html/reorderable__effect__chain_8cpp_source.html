157 void ReorderableEffectChain::processWithInput(const poly_float* audio_in, int num_samples) {
158 mono_float float_order = std::round(input(kOrder)->at(0)[0]);
159 if (float_order != last_order_)
160 utils::decodeFloatToOrder(effect_order_, float_order, constants::kNumEffects);
161 last_order_ = float_order;
162
163 for (int i = 0; i < constants::kNumEffects; ++i) {
164 VITAL_ASSERT(utils::isFinite(audio_in, num_samples));
165
166 int index = effect_order_[i];
167 bool on = effects_on_[index]->value();
168 bool enabled = effects_[index]->enabled();
169 if (on != enabled)
170 effects_[index]->enable(on);
171
172 if (on) {
173 effects_[index]->processWithInput(audio_in, num_samples);
174 audio_in = effects_[index]->output(0)->buffer;
175 }
176 }
177
178 VITAL_ASSERT(utils::isFinite(audio_in, num_samples));
179 utils::copyBuffer(output()->buffer, audio_in, num_samples);
180 }
