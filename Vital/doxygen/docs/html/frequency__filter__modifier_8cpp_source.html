2Summary:
3FrequencyFilterModifier is a wavetable component that modifies the harmonic content of a wave by applying frequency-domain filters like low-pass, band-pass, high-pass, or comb filtering. FrequencyFilterModifierKeyframe stores parameters (cutoff, shape) that define how frequencies are allowed or attenuated. By interpolating between keyframes and optionally normalizing the result, this component enables flexible and expressive frequency shaping of wavetables.

15 constexpr float kMaxPower = 9.0f;

18 force_inline double powerScale(double value, double power) {
19 // Applies a nonlinear scaling function to 'value', controlled by 'power'.
20 // Used to shape the comb filter pattern or other transformations.
21 static constexpr float kMinPower = 0.01f;
22 if (fabs(power) < kMinPower)
23 return value;
24
25 double abs_value = fabs(value);
26
27 double numerator = exp(power * abs_value) - 1.0f;
28 double denominator = exp(power) - 1.0f;
29 if (value >= 0.0f)
30 return numerator / denominator;
31 return -numerator / denominator;
32 }
33
34 force_inline float combWave(float t, float power) {
35 // Generates a comb-like pattern based on the input time 't' and 'power'.
36 // This pattern repeats and creates a unique frequency-domain shape for the comb filter mode.
37 float range = t - floorf(t);
38 return 2.0f * powerScale(1.0f - fabsf(2.0f * range - 1.0f), power);
39 }

66void FrequencyFilterModifier::FrequencyFilterModifierKeyframe::render(vital::WaveFrame* wave_frame) {
67 // Apply the multiplier to each frequency bin in the wave_frame's frequency domain representation.
68 for (int i = 0; i < vital::WaveFrame::kNumRealComplex; ++i)
69 wave_frame->frequency_domain[i] *= getMultiplier(i);
70
71 wave_frame->toTimeDomain();
72
73 // Optionally normalize the wave after applying the filter.
74 if (normalize_) {
75 wave_frame->normalize(true);
76 wave_frame->toFrequencyDomain();
77 }

93float FrequencyFilterModifier::FrequencyFilterModifierKeyframe::getMultiplier(float index) {
94 // Convert cutoff parameter to a frequency index and then shape the response.
95 float cutoff_index = std::pow(2.0f, cutoff_);
96 float cutoff_delta = index - cutoff_index;
97
98 float slope = 1.0f / vital::utils::interpolate(1.0f, kMaxSlopeReach, shape_ * shape_);
99 float power = vital::utils::interpolate(kMinPower, kMaxPower, shape_);
100
101 // Based on style, determine how frequencies above/below cutoff are attenuated.
102 if (style_ == kLowPass)
103 return vital::utils::clamp(1.0f - slope * cutoff_delta, 0.0f, 1.0f);
104 if (style_ == kBandPass)
105 return vital::utils::clamp(1.0f - fabsf(slope * cutoff_delta), 0.0f, 1.0f);
106 if (style_ == kHighPass)
107 return vital::utils::clamp(1.0f + slope * cutoff_delta, 0.0f, 1.0f);
108 if (style_ == kComb)
109 return combWave(index / (cutoff_index * 2.0f), power);
110
111 return 0.0f;
112}
